One Time Password

let inputs = Array.from(document.querySelectorAll('.otp-inputs input'));
inputs.forEach(input => {
	input.addEventListener('keydown', e => {
		let idx = inputs.indexOf(input);
		if(e.code === 'Delete'){
			input.value = '';
		}else if(e.code === 'Backspace'){
			input.value = '';
			if(idx !== 0){
				inputs[idx - 1].focus();
			}
		}else if(e.keyCode <= 57 && e.keyCode >= 48){
			input.value = e.key;
			if(idx !== inputs.length - 1){
				inputs[idx + 1].focus();
			}
		}else if(e.code === 'ArrowRight'){
			if(idx !== inputs.length - 1){
				inputs[idx + 1].focus();
			}
		}else if(e.code === 'ArrowLeft'){
			if(idx !== 0){
				inputs[idx - 1].focus();
			}
		}
		e.preventDefault();
		return false;
	});
});
